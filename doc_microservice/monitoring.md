```yaml
groups:
- name: eventstream
  rules:
  
  # High latency alert
  - alert: EventStreamHighLatency
    expr: histogram_quantile(0.99, eventstream_latency_ns_bucket) > 10000000
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "High P99 latency detected"
      description: "P99 latency is {{ $value | humanizeDuration }}"
  
  # Queue near capacity
  - alert: EventStreamQueueFull
    expr: eventstream_queue_depth / eventstream_queue_capacity > 0.8
    for: 2m
    labels:
      severity: critical
    annotations:
      summary: "Queue {{ $labels.queue }} is near capacity"
      description: "Queue is at {{ $value | humanizePercentage }}"
  
  # Node down
  - alert: EventStreamNodeDown
    expr: up{job="eventstream"} == 0
    for: 1m
    labels:
      severity: critical
    annotations:
      summary: "EventStream node {{ $labels.instance }} is down"
  
  # No leader
  - alert: EventStreamNoLeader
    expr: sum(eventstream_raft_state{state="leader"}) == 0
    for: 30s
    labels:
      severity: critical
    annotations:
      summary: "No Raft leader elected"
  
  # High replication lag
  - alert: EventStreamReplicationLag
    expr: eventstream_raft_replication_lag > 100
    for: 5m
    labels:
      severity: warning
    annotations:
      summary: "High replication lag on {{ $labels.peer }}"

  # Events dropped
```
