### SpscRingBuffer

```cpp
#include "core/queue/SpscRingBuffer.hpp"

// Create queue (16K capacity)
SpscRingBuffer<EventPtr, 16384> queue;

// Producer thread
void producer() {
    while (running) {
        auto event = pool.acquire();
        event->topic = "data";
        
        if (!queue.push(std::move(event))) {
            // Queue full - backpressure
            std::this_thread::yield();
        }
    }
}

// Consumer thread
void consumer() {
    EventPtr event;
    while (running) {
        if (queue.pop(event)) {
            process(event);
        } else {
            // Queue empty
            std::this_thread::yield();
        }
    }
}
```

### MpscQueue

```cpp
#include "core/queue/MpscQueue.hpp"

// Create queue (64K capacity)
MpscQueue<EventPtr, 65536> queue;

// Multiple producer threads (thread-safe!)
void producer(int id) {
    while (running) {
        auto event = pool.acquire();
        event->header.id = generateId(id);
        
        while (!queue.push(std::move(event))) {
            std::this_thread::yield();
        }
    }
}

// Single consumer thread
void consumer() {
    EventPtr event;
    while (running) {
        if (queue.pop(event)) {
            dispatch(event);
        }
    }
}

// Launch
std::thread c(consumer);
std::thread p1(producer, 1);
std::thread p2(producer, 2);
std::thread p3(producer, 3);
```
